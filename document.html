dplyr::group_by

import a small dataset: mtcars

  • group_by(): perform group-wise calculations
mtcars1 <- read.csv(file = "C:\\data\\mtcars.csv", 
                    stringsAsFactors = FALSE) # import the dataset

# convert mpg to kml ...
df16 <- mtcars1 %>%
  group_by(
    cyl
  ) %>%
  summarise( # within each CYLINDER CLASS ...
    ave_mpg  = mean(mpg, na.rm = TRUE), # what is the average MPG?
    max_disp = max(disp, na.rm = TRUE), # what is the maximum displacement?
    min_wt   = min(wt, na.rm = TRUE)    # what is the minimum weight?
  ) %>% ungroup()

dplyr::group_by

import a large(ish) dataset: opioids prescription 2013

  • group_by(): perform group-wise calculations
filename <- file.path("C:\\data\\opioid_prescribers_data2013.csv") # file location
op_pres13 <- read.csv(file = filename, stringsAsFactors = FALSE) # import the dataset

# What proportion of Total Medicare Claims did the Doctor handle?
df17  <- op_pres13 %>%
  group_by(
    Prscrbr_State_Abrvtn
  ) %>%
  mutate( # Compute Total Claims by State
    Tot_Clms_by_State = sum(Tot_Clms)
  ) %>% ungroup() %>%
  mutate( # Divide Claims to Total Claims
    Prop_Clms_by_Pres = Tot_Clms / Tot_Clms_by_State
  )

print(paste("The maximum Medicare claims within any State by a single doctor is",
            paste0(floor(max(df17$Prop_Clms_by_Pres, na.rm = TRUE)*100), "%.")))

dataset creation

#------------------------------------------------------------------

student <- c("A", "B", "C", "F")
score1  <- c(10, 20, 30, 60)

dataset1 <- data.frame(student, score1)

#------------------------------------------------------------------

student <- c("C", "D", "E", "F")
score2  <- c(300, 400, 500, 600)

dataset2 <- data.frame(student, score2)

#------------------------------------------------------------------

merge syntax using DplyR

  • full_join()
dataset1 %>% full_join(
      dataset2, 
      by = c("student")
      ) %>% 
  arrange(student)
  • inner_join()
dataset1 %>% inner_join(
      dataset2, 
      by = c("student")
      ) %>% 
  arrange(student)
  • left_join()
dataset1 %>% left_join(
      dataset2, 
      by = c("student")
      ) %>% 
  arrange(student)
  • right_join()
dataset1 %>% right_join(
      dataset2, 
      by = c("student")
      ) %>% 
  arrange(student)
